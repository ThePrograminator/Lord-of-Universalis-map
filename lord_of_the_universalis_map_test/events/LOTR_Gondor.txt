# Random area wants more autonomy
country_event = {
	id = flavor_gondor.1
	title = flavor_gondor.1.t
	desc = flavor_gondor.1.d
	picture = LIBERUM_VETO_eventPicture

	is_triggered_only = yes
	
	trigger = {
		AND = {
			tag = X01
			
		}
		NOT = {
			any_owned_province = {
				OR = {
					has_province_modifier = province_increased_autonomy
					has_province_modifier = province_didnt_increased_autonomy
				}
			}
		}
	}
	
	#yes
	option = { 
		name = flavor_gondor.1.a
			random_owned_area = {
			limit = {
				owned_by = ROOT
			}
			if = {
				limit = {
					NOT = {
						culture_group = culture_group_orcs
						culture_group = culture_group_melkor
						culture_group = culture_group_wilderness
					}
					NOT = {
						religion_group = religion_group_melkor
						religion_group = religion_group_wilderness
					}
				}
				set_local_autonomy = 50
			
				add_province_modifier = {
					name = "province_increased_autonomy"
					duration = 1095
				}
			}
		}
	}
	
	#No
	option = { 
		name = flavor_gondor.1.b
		random_owned_area = {
			limit = {
				owned_by = ROOT
			}
			if = {
				limit = {
					NOT = {
						culture_group = culture_group_orcs
						culture_group = culture_group_melkor
						culture_group = culture_group_wilderness
					}
					NOT = {
						religion_group = religion_group_melkor
						religion_group = religion_group_wilderness
					}
				}
			
				add_province_modifier = {
					name = "province_didnt_increased_autonomy"
					duration = 1095
				}
			}
		}
	}
}

# Rohan calls for aid
country_event = {
	id = flavor_gondor.6
	title = flavor_gondor.6.t
	desc = flavor_gondor.6.d
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes
	
	trigger = {
		AND = {
			tag = X01
			X16 = {
				exists = yes
			}
			OR = {
				X16 = {
					OR = {
						vassal_of = ROOT
						march_of = ROOT
					}
				}
				alliance_with = X16
				truce_with = X16
			}
			
		}
		NOT = {
			war_with = X16
			truce_with = X16
		}
	}
	
	#yes send manpower
	option = { 
		name = flavor_gondor.6.a
		add_manpower = -5
		
		X16 = {
			add_opinion = {
				who = ROOT
				modifier = gondor_aided_rohan
				years = 25
			}
		}
	}
	
	#No keep manpower
	option = { 
		name = flavor_gondor.6.b
		
		X16 = {
			add_opinion = {
				who = ROOT
				modifier = gondor_didnt_aid_rohan
				years = 25
			}
		}
		
		hidden_effect = {
			country_event = {
				id = flavor_gondor.6
				days = 3
			}
		}
	}
}

	}
