# Increase White Mountain Production
country_event = {
	id = flavor_gondorian.3
	title = flavor_gondorian.3.t
	desc = flavor_gondorian.3.d
	picture = CITY_DEVELOPMENT_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		any_owned_province = {
			OR = {
				trade_goods = copper
				trade_goods = iron
				trade_goods = silver
				trade_goods = gold
				trade_goods = stone
				trade_goods = marble
				trade_goods = gems
				trade_goods = mithril
			}
				
			region = gondor_region
			owner = {
				technology_group = lotr_dunedain
			}
		}
	}
	
	#Yes
	option = {
		name = "flavor_gondorian.3.a"
		
		owner = {
			add_treasury = -50
			add_dip_power = -50
			add_mil_power = -25
		}
		
		custom_tooltip = white_mountain_production
		
		tooltip = {
			add_province_modifier = {
				name = province_gondorian_increased_production
				duration = 3650
			}
		}
		
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						trade_goods = copper
						trade_goods = iron
						trade_goods = silver
						trade_goods = gold
						trade_goods = stone
						trade_goods = marble
						trade_goods = gems
						trade_goods = mithril
					}
					is_core = owner
					region = gondor_region
				}
				
				add_province_modifier = {
					name = province_gondorian_increased_production
					duration = 3650
				}
			}
		}
	}
	
	#No
	option = {
		name = "flavor_gondorian.3.b"
		
		owner = {
			add_prestige = -5
			add_legitimacy = -5
		}
	}
}

# Corsair Raider
country_event = {
	id = flavor_gondorian.9
	title = "flavor_gondorian.9.title"
	desc = "flavor_gondorian.9.desc"
	picture = BORDER_TENSION_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		AND = {
			technology_group = lotr_dunedain
			exists = X27
			
		}
		any_owned_province = {
			is_core = ROOT
			OR = {
					area = harithilien_area
					area = lebenninivet_area
					area = andrast_area
					area = ethir_anduin_area
					area = anfalas_area
					area = belfalas_area
					area = linhir_area
					area = dol_amroth_area
					area = tol_falas_area
					area = edhellond_area
				}
			
		}
		NOT = {
			has_country_modifier = gondorian_spawn_umbar_raider_timer
		}
	}
	
	option = {
		name = "flavor_gondorian.9.a"
		hidden_effect = {
			add_country_modifier = {
				name = gondorian_spawn_umbar_raider_timer 
				duration = 1095
				hidden = yes
			}
		}
		
		random_owned_province = {
			limit = {
				OR = {
					area = harithilien_area
					area = lebenninivet_area
					area = andrast_area
					area = ethir_anduin_area
					area = anfalas_area
					area = belfalas_area
					area = linhir_area
					area = dol_amroth_area
					area = tol_falas_area
					area = edhellond_area
				}
				NOT = { 
					units_in_province = 1 
				}
				is_capital = no
				has_port = yes
			}
			random_list = {
				15 = {
					spawn_rebels = {
						type = corsair_raider_rebels
						size = 2
					}
				}
				75 = {
					spawn_rebels = {
						type = corsair_raider_rebels
						size = 1
					}
				}
			}
		}
	}
}
